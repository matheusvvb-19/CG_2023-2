84 void initialize() {
85 image_height = static_cast<int>(image_width / aspect_ratio);
86 image_height = (image_height < 1) ? 1 : image_height;
87
88 center = lookfrom;
89
90 // Determine viewport dimensions.
91 auto theta = degrees_to_radians(vfov);
92 auto h = tan(theta/2);
93 auto viewport_height = 2 * h * focus_dist;
94 auto viewport_width = viewport_height * (static_cast<double>(image_width)/image_height);
95
96 // Calculate the u,v,w unit basis vectors for the camera coordinate frame.
97 w = unit_vector(lookfrom - lookat);
98 u = unit_vector(cross(vup, w));
99 v = cross(w, u);
100
101 // Calculate the vectors across the horizontal and down the vertical viewport edges.
102 vec3 viewport_u = viewport_width * u; // Vector across viewport horizontal edge
103 vec3 viewport_v = viewport_height * -v; // Vector down viewport vertical edge
104
105 // Calculate the horizontal and vertical delta vectors to the next pixel.
106 pixel_delta_u = viewport_u / image_width;
107 pixel_delta_v = viewport_v / image_height;
108
109 // Calculate the location of the upper left pixel.
110 auto viewport_upper_left = center - (focus_dist * w) - viewport_u/2 - viewport_v/2;
111 pixel00_loc = viewport_upper_left + 0.5 * (pixel_delta_u + pixel_delta_v);
112
113 // Calculate the camera defocus disk basis vectors.
114 auto defocus_radius = focus_dist * tan(degrees_to_radians(defocus_angle / 2));
115 defocus_disk_u = u * defocus_radius;
116 defocus_disk_v = v * defocus_radius;
117 }

126 ray get_ray(int i, int j) const {
127 auto pixel_center = pixel00_loc + (i * pixel_delta_u) + (j * pixel_delta_v);
128 auto pixel_sample = pixel_center + pixel_sample_square();
129
130 auto ray_origin = (defocus_angle <= 0) ? center : defocus_disk_sample();
131 auto ray_direction = pixel_sample - ray_origin;
132
133 return ray(ray_origin, ray_direction);
134 }

176 color ray_color(const ray& r, int depth, const hittable& world) const {
177 if (depth <= 0)
178 return color(0,0,0);
179
180 hit_record rec;
181
182 if (world.hit(r, interval(0.001, infinity), rec)) {
183 ray scattered;
184 color attenuation;
185 if (rec.mat->scatter(r, rec, attenuation, scattered))
186 return attenuation * ray_color(scattered, depth-1, world);
187
188 return color(0,0,0);
189 }
190
191 vec3 unit_direction = unit_vector(r.direction());
192 auto a = 0.5*(unit_direction.y() + 1.0);
193 return (1.0-a)*color(1.0, 1.0, 1.0) + a*color(0.5, 0.7, 1.0);
194 }
