138 void lerArquivoObj(const std::string& filename) {
139 std::ifstream file(filename);
140
141 std::string line;
142 std::regex regexVertex("^v (-?\\d+\\.\\d+) (-?\\d+\\.\\d+) (-?\\d+\\.\\d+)");
143 std::regex regexTexCoord("^vt (-?\\d+\\.\\d+) (-?\\d+\\.\\d+)");
144 std::regex regexNormal("^vn (-?\\d+\\.\\d+) (-?\\d+\\.\\d+) (-?\\d+\\.\\d+)");
145 std::regex regexFace("^f (\\d+)/(\\d+)/(\\d+) (\\d+)/(\\d+)/(\\d+) (\\d+)/(\\d+)/(\\d+)");
146
147 while (std::getline(file, line)) {
148 if (std::regex_match(line, regexVertex)) {
149 Vertice vertex;
150 sscanf(line.c_str(), "v %f %f %f", &vertex.x, &vertex.y, &vertex.z);
151 vertices.push_back(vertex);
152
153 } else if (std::regex_match(line, regexTexCoord)) {
154 CoordenadaTextura texCoord;
155 sscanf(line.c_str(), "vt %f %f", &texCoord.u, &texCoord.v);
156 coordenadasTextura.push_back(texCoord);
157
158 } else if (std::regex_match(line, regexNormal)) {
159 Normal normal;
160 sscanf(line.c_str(), "vn %f %f %f", &normal.x, &normal.y, &normal.z);
161 normais.push_back(normal);
162
163 } else if (std::regex_match(line, regexFace)) {
164 std::vector<VerticesDeFace> faceVertices;
165 std::smatch matches;
166 if (std::regex_search(line, matches, regexFace)) {
167 for (size_t i = 1; i < matches.size(); i += 3) {
168 VerticesDeFace faceVertex;
169 faceVertex.IndiceVertice = std::stoi(matches[i]);
170 faceVertex.IndiceTextura = std::stoi(matches[i + 1]);
171 faceVertex.IndiceNormal = std::stoi(matches[i + 2]);
172 faceVertices.push_back(faceVertex);
173 }
174 }
175 faces.push_back(faceVertices);
176 }
177 }
178
179 file.close();
180 }
