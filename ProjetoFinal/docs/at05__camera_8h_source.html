88 void initialize() {
89 image_height = static_cast<int>(image_width / aspect_ratio);
90 image_height = (image_height < 1) ? 1 : image_height;
91
92 center = lookfrom;
93
94 // Determine viewport dimensions.
95 auto theta = degrees_to_radians(vfov);
96 auto h = tan(theta/2);
97 auto viewport_height = 2 * h * focus_dist;
98 auto viewport_width = viewport_height * (static_cast<double>(image_width)/image_height);
99
100 // Calculate the u,v,w unit basis vectors for the camera coordinate frame.
101 w = unit_vector(lookfrom - lookat);
102 u = unit_vector(cross(vup, w));
103 v = cross(w, u);
104
105 // Calculate the vectors across the horizontal and down the vertical viewport edges.
106 vec3 viewport_u = viewport_width * u; // Vector across viewport horizontal edge
107 vec3 viewport_v = viewport_height * -v; // Vector down viewport vertical edge
108
109 // Calculate the horizontal and vertical delta vectors to the next pixel.
110 pixel_delta_u = viewport_u / image_width;
111 pixel_delta_v = viewport_v / image_height;
112
113 // Calculate the location of the upper left pixel.
114 auto viewport_upper_left = center - (focus_dist * w) - viewport_u/2 - viewport_v/2;
115 pixel00_loc = viewport_upper_left + 0.5 * (pixel_delta_u + pixel_delta_v);
116
117 // Calculate the camera defocus disk basis vectors.
118 auto defocus_radius = focus_dist * tan(degrees_to_radians(defocus_angle / 2));
119 defocus_disk_u = u * defocus_radius;
120 defocus_disk_v = v * defocus_radius;
121 }

130 ray get_ray(int i, int j) const {
131 auto pixel_center = pixel00_loc + (i * pixel_delta_u) + (j * pixel_delta_v);
132 auto pixel_sample = pixel_center + pixel_sample_square();
133
134 auto ray_origin = (defocus_angle <= 0) ? center : defocus_disk_sample();
135 auto ray_direction = pixel_sample - ray_origin;
136
137 return ray(ray_origin, ray_direction);
138 }

180 color ray_color(const ray& r, int depth, const hittable& world) const {
181 if (depth <= 0)
182 return color(0,0,0);
183
184 hit_record rec;
185
186 if (world.hit(r, interval(0.001, infinity), rec)) {
187 ray scattered;
188 color attenuation;
189 if (rec.mat->scatter(r, rec, attenuation, scattered))
190 return attenuation * ray_color(scattered, depth-1, world);
191
192 return color(0,0,0);
193 }
194
195 vec3 unit_direction = unit_vector(r.direction());
196 auto a = 0.5*(unit_direction.y() + 1.0);
197 return (1.0-a)*color(1.0, 1.0, 1.0) + a*color(0.5, 0.7, 1.0);
198 }
