99
100 vec3 unit_direction = unit_vector(r_in.direction());
101 double cos_theta = fmin(dot(-unit_direction, rec.normal), 1.0);
102 double sin_theta = sqrt(1.0 - cos_theta*cos_theta);
103
104 bool cannot_refract = refraction_ratio * sin_theta > 1.0;
105 vec3 direction;
106
107 if (cannot_refract || reflectance(cos_theta, refraction_ratio) > random_double())
108 direction = reflect(unit_direction, rec.normal);
109 else

111
112 scattered = ray(rec.p, direction);
