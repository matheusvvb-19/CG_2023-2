148 void lerArquivoObj(const std::string& filename) {
149 std::ifstream file(filename);
150
151 std::string line;
152 std::regex regexVertex("^v (-?\\d+\\.\\d+) (-?\\d+\\.\\d+) (-?\\d+\\.\\d+)");
153 std::regex regexTexCoord("^vt (-?\\d+\\.\\d+) (-?\\d+\\.\\d+)");
154 std::regex regexNormal("^vn (-?\\d+\\.\\d+) (-?\\d+\\.\\d+) (-?\\d+\\.\\d+)");
155 std::regex regexFace("^f (\\d+)/(\\d+)/(\\d+) (\\d+)/(\\d+)/(\\d+) (\\d+)/(\\d+)/(\\d+)");
156
157 while (std::getline(file, line)) {
158 if (std::regex_match(line, regexVertex)) {
159 Vertice vertex;
160 sscanf(line.c_str(), "v %f %f %f", &vertex.x, &vertex.y, &vertex.z);
161 vertices.push_back(vertex);
162
163 } else if (std::regex_match(line, regexTexCoord)) {
164 CoordenadaTextura texCoord;
165 sscanf(line.c_str(), "vt %f %f", &texCoord.u, &texCoord.v);
166 coordenadasTextura.push_back(texCoord);
167
168 } else if (std::regex_match(line, regexNormal)) {
169 Normal normal;
170 sscanf(line.c_str(), "vn %f %f %f", &normal.x, &normal.y, &normal.z);
171 normais.push_back(normal);
172
173 } else if (std::regex_match(line, regexFace)) {
174 std::vector<VerticesDeFace> faceVertices;
175 std::smatch matches;
176 if (std::regex_search(line, matches, regexFace)) {
177 for (size_t i = 1; i < matches.size(); i += 3) {
178 VerticesDeFace faceVertex;
179 faceVertex.IndiceVertice = std::stoi(matches[i]);
180 faceVertex.IndiceTextura = std::stoi(matches[i + 1]);
181 faceVertex.IndiceNormal = std::stoi(matches[i + 2]);
182 faceVertices.push_back(faceVertex);
183 }
184 }
185 faces.push_back(faceVertices);
186 }
187 }
188
189 file.close();
190 };
